Example Usage

Here's how to validate an AI output using our service:

import requests

# Your AI output data and schema
validation_request = {
    "data": {
        "product_name": "Smart Home Hub",
        "description": "A central control device for all your smart home needs",
        "price": 149.99,
        "in_stock": true
    },
    "schema": {
        "product_name": {"type": "string", "required": true},
        "description": {"type": "string", "required": true},
        "price": {"type": "number", "required": true},
        "in_stock": {"type": "boolean", "required": false}
    },
    "type": "product",
    "level": "standard"
}

# Send to validation service
response = requests.post(
    "http://your-service-url/validate",
    headers={"x-api-key": "your-secret-key"},  # Only if authentication is enabled
    json=validation_request
)

# Process validation results
result = response.json()
print(f"Validation result: {result['is_valid']}")
